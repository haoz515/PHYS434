X = np.sort(data)
clear = np.array([])
throw = np.array([])

def truthtable(sigma):
    sig = stats.norm.sf(sigma)
    max_t = stats.norm.isf(sig, loc=12, scale=0.4)
    min_t = stats.norm.isf(1-sig, loc=12, scale=0.4)
    
    count1 = 0
    count2 = 0
    for x in data:
        if(x > min_t and x < max_t): count1 = count1 + 1
        elif(x < min_t or x > max_t): count2= count2 + 1

    print("thrown values' # :",count2)
    print("# of elements after cleared:", count1)
    
    clear = X[np.logical_and(X > min_t, X < max_t)]
    throw = X[np.logical_or(X < min_t, X > max_t)]
    true_pos = sum((Counter(pure) & Counter(clear)).values())
    false_pos = sum((Counter(clear) & Counter(noise)).values())
    true_neg = sum((Counter(pure) & Counter(throw)).values())
    false_neg = sum((Counter(throw) & Counter(noise)).values())
    
    table = pd.DataFrame(data=[[true_pos,false_pos],[true_neg,false_neg]],
                     index=["Positive", "Negative"],
                     columns=["True", "False"])
    return table
